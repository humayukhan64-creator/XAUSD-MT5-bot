//| RSI PRO BOT (FINAL CLEAN VERSION)                               |
//+------------------------------------------------------------------+
#property strict

#include <Trade/Trade.mqh>
CTrade trade;

// ===== INPUTS =====
input double LotSize = 0.10;
input int RSIPeriod = 14;

input double BuyLevel = 30;
input double SellLevel = 70;

input double ConfirmBuy = 31.5;
input double ConfirmSell = 68.5;

input double MomentumThreshold = 0.20;

input int RSI_Filter_Candles = 5;
input double RSI_Flat_Threshold = 1.0;

input int TrailStart = 120;
input int TrailStep  = 60;

// SL SETTINGS
input int ATR_Period = 14;
input double MinSL_Dollar = 15;
input double MaxSL_Dollar = 25;

// Filters
input int CooldownMinutes = 5;

// ===== GLOBAL =====
int rsiHandle, atrHandle;
datetime lastTradeTime = 0;
datetime lastCandleTime = 0;

bool oversoldTouched = false;
bool overboughtTouched = false;

//+------------------------------------------------------------------+
int OnInit()
{
   rsiHandle = iRSI(_Symbol, PERIOD_M1, RSIPeriod, PRICE_CLOSE);
   atrHandle = iATR(_Symbol, PERIOD_M1, ATR_Period);
   return(INIT_SUCCEEDED);
}
//+------------------------------------------------------------------+
void OnTick()
{
   // ===== One trade per candle =====
   datetime currentCandle = iTime(_Symbol, PERIOD_M1, 0);
   bool isNewCandle = (currentCandle != lastCandleTime);

   if(isNewCandle)
      lastCandleTime = currentCandle;

   // ===== Cooldown =====
   if(TimeCurrent() - lastTradeTime < CooldownMinutes * 60)
      return;

   // ===== RSI =====
   double rsi[];
   ArraySetAsSeries(rsi, true);
   int copied = CopyBuffer(rsiHandle, 0, 0, RSI_Filter_Candles + 2, rsi);
   if(copied <= RSI_Filter_Candles + 1)
      return;

   double rsi1 = rsi[1];
   double rsi2 = rsi[2];

   // ===== ATR =====
   double atr[];
   if(CopyBuffer(atrHandle, 0, 0, 1, atr) <= 0)
      return;

   double atrValue = atr[0];

   // ===== RSI FILTER =====
   bool flatRSI = true;

   for(int i = 1; i <= RSI_Filter_Candles; i++)
   {
      if(MathAbs(rsi[i] - rsi[i+1]) > RSI_Flat_Threshold)
      {
         flatRSI = false;
         break;
      }
   }

   if(flatRSI && MathAbs(rsi[1] - rsi[RSI_Filter_Candles]) < RSI_Flat_Threshold * 2)
      return;

   // ===== RESET (FIXED FINAL) =====
   if(rsi1 > ConfirmBuy && rsi2 > ConfirmBuy)
      oversoldTouched = false;

   if(rsi1 < ConfirmSell && rsi2 < ConfirmSell)
      overboughtTouched = false;

   // ===== TRIGGER =====
   if(rsi1 < BuyLevel)
      oversoldTouched = true;

   if(rsi1 > SellLevel)
      overboughtTouched = true;

   // ===== ENTRY =====
   if(!PositionSelect(_Symbol))
   {
      double tickValue = SymbolInfoDouble(_Symbol, SYMBOL_TRADE_TICK_VALUE);
      double tickSize  = SymbolInfoDouble(_Symbol, SYMBOL_TRADE_TICK_SIZE);

      // BUY
      if(oversoldTouched &&
         rsi2 < ConfirmBuy &&
         rsi1 > ConfirmBuy &&
         (rsi1 - rsi2) > MomentumThreshold)
      {
         double slDistance = atrValue * 2;
         double slDollar = slDistance / tickSize * tickValue * LotSize;

         if(slDollar > 0)
         {
            if(slDollar < MinSL_Dollar)
               slDistance *= (MinSL_Dollar / slDollar);

            if(slDollar > MaxSL_Dollar)
               slDistance *= (MaxSL_Dollar / slDollar);

            double sl = SymbolInfoDouble(_Symbol, SYMBOL_BID) - slDistance;

            trade.Buy(LotSize, _Symbol, 0, sl, 0);

            oversoldTouched = false;
            lastTradeTime = TimeCurrent();
         }
      }

      // SELL
      if(overboughtTouched &&
         rsi2 > ConfirmSell &&
         rsi1 < ConfirmSell &&
         (rsi2 - rsi1) > MomentumThreshold)
      {
         double slDistance = atrValue * 2;
         double slDollar = slDistance / tickSize * tickValue * LotSize;

         if(slDollar > 0)
         {
            if(slDollar < MinSL_Dollar)
               slDistance *= (MinSL_Dollar / slDollar);

            if(slDollar > MaxSL_Dollar)
               slDistance *= (MaxSL_Dollar / slDollar);

            double sl = SymbolInfoDouble(_Symbol, SYMBOL_ASK) + slDistance;

            trade.Sell(LotSize, _Symbol, 0, sl, 0);

            overboughtTouched = false;
            lastTradeTime = TimeCurrent();
         }
      }
   }

   // ===== FAST EXIT =====
   bool hasPos = PositionSelect(_Symbol);

   if(hasPos)
 {
   long type = PositionGetInteger(POSITION_TYPE);

   if(type == POSITION_TYPE_BUY && rsi1 < (BuyLevel - 1))
      trade.PositionClose(_Symbol);

   if(type == POSITION_TYPE_SELL && rsi1 > (SellLevel + 1))
      trade.PositionClose(_Symbol);
}

   // ===== TRAILING =====
if(PositionSelect(_Symbol))
{
   double open = PositionGetDouble(POSITION_PRICE_OPEN);
   double sl   = PositionGetDouble(POSITION_SL);

   double bid = SymbolInfoDouble(_Symbol, SYMBOL_BID);
   double ask = SymbolInfoDouble(_Symbol, SYMBOL_ASK);

   // BUY
   if(PositionGetInteger(POSITION_TYPE)==POSITION_TYPE_BUY)
   {
      double profit = (bid - open) / _Point;

      if(profit > TrailStart)
      {
         double lvl = MathFloor((profit - TrailStart) / TrailStep);
         double newSL = open + lvl * TrailStep * _Point;

         if(newSL > sl)
            trade.PositionModify(_Symbol, newSL, 0);
      }
   }

   // SELL
   if(PositionGetInteger(POSITION_TYPE)==POSITION_TYPE_SELL)
   {
      double profit = (open - ask) / _Point;

      if(profit > TrailStart)
      {
         double lvl = MathFloor((profit - TrailStart) / TrailStep);
         double newSL = open - lvl * TrailStep * _Point;

         if(newSL < sl || sl == 0)
            trade.PositionModify(_Symbol, newSL, 0);
      }
   }
}
}
